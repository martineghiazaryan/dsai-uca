7. Retrieve the mails 

PREFIX : <http://www.universite-cote-d'azur.org/stuff#>
SELECT ?name ?email
WHERE {
  ?teacher a :Teacher ;
           :teacherName ?name ;
           :emailAddress ?email .
}


8. Retrieve the names and dates of birth of all the students who were born after January 1st, 2000

PREFIX : <http://www.universite-cote-d'azur.org/stuff#>
SELECT ?name ?dob
WHERE {
  ?student rdf:type :Student ;
           :studentName ?name ;
           :dateOfBirth ?dob .
  FILTER (?dob > "2000-01-01"^^xsd:date)
}


9.Retrieve the names and popularity of all the teachers who specialize in teaching Mathematics

PREFIX : <http://www.universite-cote-d'azur.org/stuff#>
SELECT ?name ?popularity
WHERE {
  ?teacher rdf:type :Teacher ;
           :teacherName ?name ;
           :teachingSubject "Mathematics" ;
           :teacherPopularity ?popularity .
}


10. Retrieve the names and scholarship types of all the students who received a scholarship

PREFIX : <http://www.universite-cote-d'azur.org/stuff#>
SELECT ?name ?promotion
WHERE {
  ?student rdf:type :Student ;
           :studentName ?name .
  OPTIONAL {?student :receivesPromotion ?promotion}
}

11. Retrieve the names and genders of all the female students



PREFIX : <http://www.universite-cote-d'azur.org/stuff#>
SELECT ?name ?gender
WHERE {
  ?student rdf:type :Student ;
           :studentName ?name ;
           :studentGender ?gender .
  FILTER (?gender = "Female")
}
